006public class Item implements Comparable
007{
008    private String aNom;
009    private int    aPrix;
010
011    public Item( final String pNom, final int pPrix )
012    {
013        this.aNom  = pNom;
014        this.aPrix = pPrix;
015    } // Item(..)
016    
017    public String getNom()
018    {
019        return this.aNom;
020    } // getNom()
021    
022    public int getPrix()
023    {
024        return this.aPrix;
025    } // getPrix()
026
027    @Override
028    public String toString()
029    {
030        return this.aNom + " (" + this.aPrix + "â‚¬)";
031    }
032    
033    public int compareTo( final Object pItem )
034    {
035        Integer vPrice = this.aPrix;
036        Item vItem = (Item)pItem;
037        
038        if(vPrice.compareTo(vItem.getPrix()) == 0){
039            return this.aNom.compareTo(vItem.getNom());
040        }else{
041            return vPrice.compareTo(vItem.getPrix());
042        }
043    }
044    
045    public boolean equals( final Item pItem )
046    {
047        return this.aPrix == pItem.getPrix() && this.aNom == pItem.getNom();
048    }
049} // Item
